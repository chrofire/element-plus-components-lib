import { reactive, ref } from 'vue'

const BaseRadioGroupRef = ref(null)

const info = reactive({
    ref: 'BaseRadioGroupRef',
    modelValue: '',
    'onUpdate:modelValue': value => info.modelValue = value,

        }

</script>

optionProps #

optionProps 用于 转换 传入的 options, 转换后的内容作为 ElRadio / ElRadioButton 组件的 props

""
vue
<template>
    <BaseRadioGroup v-bind="info"></BaseRadioGroup>
    <pre>{{ JSON.stringify(info.modelValue, null, 4) }}</pre>
</template>

<script setup>
import { reactive } from 'vue'
import { BaseRadioGroup } from 'element-plus-components-lib'

const info = reactive({
    ref: value => info.instanceRef = value,
    instanceRef: null,
    modelValue: '',
    'onUpdate:modelValue': value => info.modelValue = value,
    // optionType: 'button',
    options: Array.from({ length: 10 }, (_, i) => {
        return {
            label: `label-${i + 1}`
        }
    }),
    optionProps: rawOption => {
        const { label } = rawOption
        return {
            ...rawOption,
            key: label,
            label: `custom-${label}`
        }
    }
})
</script>

插槽 #

优先级

rawOption.labelRenderer > props.labelRenderer > label slot > 转换后的 option.label

""
vue
<template>
    <BaseRadioGroup v-bind="info"></BaseRadioGroup>
    <pre>{{ JSON.stringify(info.modelValue, null, 4) }}</pre>
</template>

<script lang="jsx" setup>
import { reactive } from 'vue'
import { BaseRadioGroup } from 'element-plus-components-lib'

const info = reactive({
    modelValue: '',
    'onUpdate:modelValue': value => info.modelValue = value,
    // optionType: 'button',
    options: Array.from({ length: 10 }, (_, i) => {
        if (i === 0) {
            return {
                label: `label-${i + 1}`,
                labelRenderer: (option, rawOption) => {
                    return <span class="text-red">{option.label}</span>
                }
            }
        }
        return {
            label: `label-${i + 1}`
        }
    }),
    // option组件 的 props, label 作为 结果值
    optionProps: rawOption => {
        const { label } = rawOption
        return {
            ...rawOption,
            key: label,
            label: `result-${label}`
        }
    },
    // option选项右侧 显示的内容
    labelRenderer: (option, rawOption) => {
        return `renderer-${option.label}`
    }
})
</script>
